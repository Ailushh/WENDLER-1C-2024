366void ReadApp(uint8_t * data, uint8_t length){
367
368 uint8_t i = 1;
369
370 //Recibo los días de atención en formato Sun, Mon, Tue, Wed, Thu, Fri, Sat. Si coincide con el día actual, significa que hoy habrá atención
371 if(data[0]==day[0] && data[1]==day[1]){
372 today=true;
373 }
374
375 else{
376 today=false;
377 }
378
379 if(data[0]=='I'){
380 hours_start = 0;
381 while(data[i] != 'I'){
382 hours_start = hours_start * 10;
383 hours_start = hours_start + (data[i] - '0');
384 i++;
385 }
386 }
387
388 if(data[0]=='S'){
389 minutes_start = 0;
390 while(data[i] != 'S'){
391 minutes_start = minutes_start * 10;
392 minutes_start = minutes_start + (data[i] - '0');
393 i++;
394 }
395 }
396
397 if(data[0]=='F'){
398 hours_end = 0;
399 while(data[i] != 'F'){
400 hours_end = hours_end * 10;
